37def _usable_dir(path): 

38 """ 

39 Returns True iff the desired path can be used as database path because 

40 either the directory exists and can be used, or its root directory can 

41 be used and we can make the directory as needed. 

42 """ 

43 while not os.path.exists(path): 

44 # Loop terminates because the root dir ('/' on unix) always exists. 

45 path = os.path.dirname(path) 

46 return os.path.isdir(path) and os.access(path, os.R_OK | os.W_OK | os.X_OK)

98class ExampleDatabase(metaclass=_EDMeta): 

99 """An abstract base class for storing examples in Hypothesis' internal format. 

100 

101 An ExampleDatabase maps each ``bytes`` key to many distinct ``bytes`` 

102 values, like a ``Mapping[bytes, AbstractSet[bytes]]``. 

103 """ 

104 

105 @abc.abstractmethod 

106 def save(self, key: bytes, value: bytes) -> None: 

107 """Save ``value`` under ``key``. 

108 

109 If this value is already present for this key, silently do nothing. 

110 """ 

111 raise NotImplementedError(f"{type(self).__name__}.save") 

112 

113 @abc.abstractmethod 

114 def fetch(self, key: bytes) -> Iterable[bytes]: 

115 """Return an iterable over all values matching this key.""" 

116 raise NotImplementedError(f"{type(self).__name__}.fetch") 

117 

118 @abc.abstractmethod 

119 def delete(self, key: bytes, value: bytes) -> None: 

120 """Remove this value from this key. 

121 

122 If this value is not present, silently do nothing. 

123 """ 

124 raise NotImplementedError(f"{type(self).__name__}.delete") 

125 

126 def move(self, src: bytes, dest: bytes, value: bytes) -> None: 

127 """Move ``value`` from key ``src`` to key ``dest``. Equivalent to 

128 ``delete(src, value)`` followed by ``save(src, value)``, but may 

129 have a more efficient implementation. 

130 

131 Note that ``value`` will be inserted at ``dest`` regardless of whether 

132 it is currently present at ``src``. 

133 """ 

134 if src == dest: 

135 self.save(src, value) 

136 return 

137 self.delete(src, value) 

138 self.save(dest, value)

141class InMemoryExampleDatabase(ExampleDatabase): 

142 """A non-persistent example database, implemented in terms of a dict of sets. 

143 

144 This can be useful if you call a test function several times in a single 

145 session, or for testing other database implementations, but because it 

146 does not persist between runs we do not recommend it for general use. 

147 """ 

148 

149 def __init__(self): 

150 self.data = {} 

151 

152 def __repr__(self) -> str: 

153 return f"InMemoryExampleDatabase({self.data!r})" 

154 

155 def fetch(self, key: bytes) -> Iterable[bytes]: 

156 yield from self.data.get(key, ()) 

157 

158 def save(self, key: bytes, value: bytes) -> None: 

159 self.data.setdefault(key, set()).add(bytes(value)) 

160 

161 def delete(self, key: bytes, value: bytes) -> None: 

162 self.data.get(key, set()).discard(bytes(value)) 

163

165def _hash(key): 

166 return sha384(key).hexdigest()[:16]

169class DirectoryBasedExampleDatabase(ExampleDatabase): 

170 """Use a directory to store Hypothesis examples as files. 

171 

172 Each test corresponds to a directory, and each example to a file within that 

173 directory. While the contents are fairly opaque, a 

174 ``DirectoryBasedExampleDatabase`` can be shared by checking the directory 

175 into version control, for example with the following ``.gitignore``:: 

176 

177 # Ignore files cached by Hypothesis... 

178 .hypothesis/* 

179 # except for the examples directory 

180 !.hypothesis/examples/ 

181 

182 Note however that this only makes sense if you also pin to an exact version of 

183 Hypothesis, and we would usually recommend implementing a shared database with 

184 a network datastore - see :class:`~hypothesis.database.ExampleDatabase`, and 

185 the :class:`~hypothesis.database.MultiplexedDatabase` helper. 

186 """ 

187 

188 def __init__(self, path: str) -> None: 

189 self.path = path 

190 self.keypaths = {} # type: dict 

191 

192 def __repr__(self) -> str: 

193 return f"DirectoryBasedExampleDatabase({self.path!r})" 

194 

195 def _key_path(self, key): 

196 try: 

197 return self.keypaths[key] 

198 except KeyError: 

199 pass 

200 directory = os.path.join(self.path, _hash(key)) 

201 self.keypaths[key] = directory 

202 return directory 

203 

204 def _value_path(self, key, value): 

205 return os.path.join(self._key_path(key), _hash(value)) 

206 

207 def fetch(self, key: bytes) -> Iterable[bytes]: 

208 kp = self._key_path(key) 

209 if not os.path.exists(kp): 

210 return 

211 for path in os.listdir(kp): 

212 try: 

213 with open(os.path.join(kp, path), "rb") as i: 

214 yield i.read() 

215 except OSError: 

216 pass 

217 

218 def save(self, key: bytes, value: bytes) -> None: 

219 # Note: we attempt to create the dir in question now. We 

220 # already checked for permissions, but there can still be other issues, 

221 # e.g. the disk is full 

222 mkdir_p(self._key_path(key)) 

223 path = self._value_path(key, value) 

224 if not os.path.exists(path): 

225 suffix = binascii.hexlify(os.urandom(16)).decode("ascii") 

226 tmpname = path + "." + suffix 

227 with open(tmpname, "wb") as o: 

228 o.write(value) 

229 try: 

230 os.rename(tmpname, path) 

231 except OSError: # pragma: no cover 

232 os.unlink(tmpname) 

233 assert not os.path.exists(tmpname) 

234 

235 def move(self, src: bytes, dest: bytes, value: bytes) -> None: 

236 if src == dest: 

237 self.save(src, value) 

238 return 

239 try: 

240 os.renames(self._value_path(src, value), self._value_path(dest, value)) 

241 except OSError: 

242 self.delete(src, value) 

243 self.save(dest, value) 

244 

245 def delete(self, key: bytes, value: bytes) -> None: 

246 try: 

247 os.unlink(self._value_path(key, value)) 

248 except OSError: 

249 pass

252class ReadOnlyDatabase(ExampleDatabase): 

253 """A wrapper to make the given database read-only. 

254 

255 The implementation passes through ``fetch``, and turns ``save``, ``delete``, and 

256 ``move`` into silent no-ops. 

257 

258 Note that this disables Hypothesis' automatic discarding of stale examples. 

259 It is designed to allow local machines to access a shared database (e.g. from CI 

260 servers), without propagating changes back from a local or in-development branch. 

261 """ 

262 

263 def __init__(self, db: ExampleDatabase) -> None: 

264 assert isinstance(db, ExampleDatabase) 

265 self._wrapped = db 

266 

267 def __repr__(self) -> str: 

268 return f"ReadOnlyDatabase({self._wrapped!r})" 

269 

270 def fetch(self, key: bytes) -> Iterable[bytes]: 

271 yield from self._wrapped.fetch(key) 

272 

273 def save(self, key: bytes, value: bytes) -> None: 

274 pass 

275 

276 def delete(self, key: bytes, value: bytes) -> None: 

277 pass

280class MultiplexedDatabase(ExampleDatabase): 

281 """A wrapper around multiple databases. 

282 

283 Each ``save``, ``fetch``, ``move``, or ``delete`` operation will be run against 

284 all of the wrapped databases. ``fetch`` does not yield duplicate values, even 

285 if the same value is present in two or more of the wrapped databases. 

286 

287 This combines well with a :class:`ReadOnlyDatabase`, as follows: 

288 

289 .. code-block:: python 

290 

291 local = DirectoryBasedExampleDatabase("/tmp/hypothesis/examples/") 

292 shared = CustomNetworkDatabase() 

293 

294 settings.register_profile("ci", database=shared) 

295 settings.register_profile( 

296 "dev", database=MultiplexedDatabase(local, ReadOnlyDatabase(shared)) 

297 ) 

298 settings.load_profile("ci" if os.environ.get("CI") else "dev") 

299 

300 So your CI system or fuzzing runs can populate a central shared database; 

301 while local runs on development machines can reproduce any failures from CI 

302 but will only cache their own failures locally and cannot remove examples 

303 from the shared database. 

304 """ 

305 

306 def __init__(self, *dbs: ExampleDatabase) -> None: 

307 assert all(isinstance(db, ExampleDatabase) for db in dbs) 

308 self._wrapped = dbs 

309 

310 def __repr__(self) -> str: 

311 return "MultiplexedDatabase({})".format(", ".join(map(repr, self._wrapped))) 

312 

313 def fetch(self, key: bytes) -> Iterable[bytes]: 

314 seen = set() 

315 for db in self._wrapped: 

316 for value in db.fetch(key): 

317 if value not in seen: 

318 yield value 

319 seen.add(value) 

320 

321 def save(self, key: bytes, value: bytes) -> None: 

322 for db in self._wrapped: 

323 db.save(key, value) 

324 

325 def delete(self, key: bytes, value: bytes) -> None: 

326 for db in self._wrapped: 

327 db.delete(key, value) 

328 

329 def move(self, src: bytes, dest: bytes, value: bytes) -> None: 

330 for db in self._wrapped: 

331 db.move(src, dest, value)
